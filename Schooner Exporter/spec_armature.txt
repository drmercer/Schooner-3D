int: version (1)
unsigned byte: b (number of bones - 1)
for each bone (b)
	float[3]: location of head (child end) relative to tail (parent's head)
	byte: index of parent + 1, or 0 if bone has no parent

for each action
	null-terminated UTF-8 string: action name
	byte: {moveLoc, moveRot, moveScaleAxis, moveScale} 
	unsigned byte: n (number of keyframes - 1)
	for each component of the movement (3 for position if moveLoc, 4 for quaternion if moveRot, and ((1 for scale if moveScale) or (3 for scale if moveScaleAxis))) 
				for n*3 + 1: (keyframes and control points)
					float: frame
					float: value
	for each bone (b)
		if bone has no animation,
			float: 0.0f
		else
			for each component of quaternion (w, x, y, and z)
				for n*3 + 1: (keyframes and control points)
					float: frame
					float: value
